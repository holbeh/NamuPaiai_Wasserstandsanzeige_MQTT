132 private:
133  const char *getMode(uint8_t mode) {
134  bool read = (mode & sdfat::O_READ) ? true : false;
135  bool write = (mode & sdfat::O_WRITE) ? true : false;
136  bool append = (mode & sdfat::O_APPEND) ? true : false;
137  if ( read & !write ) { return "r"; }
138  else if ( !read & write & !append ) { return "w+"; }
139  else if ( !read & write & append ) { return "a"; }
140  else if ( read & write & !append ) { return "w+"; } // may be a bug in FS::mode interpretation, "r+" seems proper
141  else if ( read & write & append ) { return "a+"; }
142  else { return "r"; }
143  }
