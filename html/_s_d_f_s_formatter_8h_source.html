70  bool clearFatDir(uint32_t bgn, uint32_t count) {
71  clearCache(false);
72  if (!card->writeStart(bgn, count)) {
73  DEBUGV("SDFS: Clear FAT/DIR writeStart failed");
74  return false;
75  }
76  esp8266::polledTimeout::periodicFastMs timeToYield(5); // Yield every 5ms of runtime
77  for (uint32_t i = 0; i < count; i++) {
78  if (timeToYield) {

105  bool writeMbr() {
106  clearCache(true);
107  sdfat::part_t* p = cache->mbr.part;
108  p->boot = 0;
109  uint16_t c = lbnToCylinder(relSector);
110  if (c > 1023) {
111  DEBUGV("SDFS: MBR CHS");
112  return false;
113  }
114  p->beginCylinderHigh = c >> 8;
115  p->beginCylinderLow = c & 0XFF;
116  p->beginHead = lbnToHead(relSector);
117  p->beginSector = lbnToSector(relSector);
118  p->type = partType;
119  uint32_t endLbn = relSector + partSize - 1;
120  c = lbnToCylinder(endLbn);
121  if (c <= 1023) {
122  p->endCylinderHigh = c >> 8;
123  p->endCylinderLow = c & 0XFF;
124  p->endHead = lbnToHead(endLbn);
125  p->endSector = lbnToSector(endLbn);
126  } else {
127  // Too big flag, c = 1023, h = 254, s = 63
128  p->endCylinderHigh = 3;
129  p->endCylinderLow = 255;
130  p->endHead = 254;
131  p->endSector = 63;
132  }
133  p->firstSector = relSector;
134  p->totalSectors = partSize;
135  if (!writeCache(0)) {
136  DEBUGV("SDFS: write MBR");
137  return false;
138  }
139  return true;
140  }

146  bool makeFat16() {
147  uint16_t const BU16 = 128;
148  uint32_t nc;
149  for (dataStart = 2 * BU16;; dataStart += BU16) {
150  nc = (cardSizeBlocks - dataStart)/sectorsPerCluster;
151  fatSize = (nc + 2 + 255)/256;
152  uint32_t r = BU16 + 1 + 2 * fatSize + 32;
153  if (dataStart < r) {
154  continue;
155  }
156  relSector = dataStart - r + BU16;
157  break;
158  }
159  // check valid cluster count for FAT16 volume
160  if (nc < 4085 || nc >= 65525) {
161  DEBUGV("SDFS: Bad cluster count");
162  }
163  reservedSectors = 1;
164  fatStart = relSector + reservedSectors;
165  partSize = nc * sectorsPerCluster + 2 * fatSize + reservedSectors + 32;
166  if (partSize < 32680) {
167  partType = 0X01;
168  } else if (partSize < 65536) {
169  partType = 0X04;
170  } else {
171  partType = 0X06;
172  }
173  // write MBR
174  if (!writeMbr()) {
175  DEBUGV("SDFS: writembr failed");
176  return false;
177  }
178 
179  clearCache(true);
180  sdfat::fat_boot_t* pb = &cache->fbs;
181  pb->jump[0] = 0XEB;
182  pb->jump[1] = 0X00;
183  pb->jump[2] = 0X90;
184  for (uint8_t i = 0; i < sizeof(pb->oemId); i++) {
185  pb->oemId[i] = ' ';
186  }
187  pb->bytesPerSector = 512;
188  pb->sectorsPerCluster = sectorsPerCluster;
189  pb->reservedSectorCount = reservedSectors;
190  pb->fatCount = 2;
191  pb->rootDirEntryCount = 512;
192  pb->mediaType = 0XF8;
193  pb->sectorsPerFat16 = fatSize;
194  pb->sectorsPerTrack = sectorsPerTrack;
195  pb->headCount = numberOfHeads;

210  DEBUGV("SDFS: FAT16 clear root failed\n");

212  }
213  clearCache(false);

215  cache->fat16[1] = 0XFFFF;
216  // write first block of FAT and backup for reserved clusters
217  if (!writeCache(fatStart) || !writeCache(fatStart + fatSize)) {
218  DEBUGV("FAT16 reserve failed");
219  return false;
220  }
221  return true;

224  bool makeFat32() {
225  uint16_t const BU32 = 8192;
226  uint32_t nc;
227  relSector = BU32;
228  for (dataStart = 2 * BU32;; dataStart += BU32) {
229  nc = (cardSizeBlocks - dataStart)/sectorsPerCluster;
230  fatSize = (nc + 2 + 127)/128;
231  uint32_t r = relSector + 9 + 2 * fatSize;
232  if (dataStart >= r) {
233  break;
234  }
235  }
236  // error if too few clusters in FAT32 volume
237  if (nc < 65525) {
238  DEBUGV("SDFS: Bad cluster count");
239  return false;
240  }
241  reservedSectors = dataStart - relSector - 2 * fatSize;
242  fatStart = relSector + reservedSectors;
243  partSize = nc * sectorsPerCluster + dataStart - relSector;
244  // type depends on address of end sector
245  // max CHS has lbn = 16450560 = 1024*255*63
246  if ((relSector + partSize) <= 16450560) {
247  // FAT32
248  partType = 0X0B;
249  } else {
250  // FAT32 with INT 13
251  partType = 0X0C;
252  }
253  if (!writeMbr()) {
254  DEBUGV("SDFS: writembr failed");
255  return false;

320  bool format(sdfat::SdFat *_fs, int8_t _csPin, SPISettings _spiSettings) {
321  card = static_cast<sdfat::Sd2Card*>(_fs->card());
322  cache = _fs->cacheClear();
323 
324  if (!card->begin(_csPin, _spiSettings)) {
325  return false;
326  }
327  cardSizeBlocks = card->cardSize();
328  if (cardSizeBlocks == 0) {
329  return false;
330  }
331 
332  cardCapacityMB = (cardSizeBlocks + 2047)/2048;
333 
334  if (cardCapacityMB <= 6) {
335  return false; // Card is too small
336  } else if (cardCapacityMB <= 16) {
337  sectorsPerCluster = 2;
338  } else if (cardCapacityMB <= 32) {
339  sectorsPerCluster = 4;
340  } else if (cardCapacityMB <= 64) {
341  sectorsPerCluster = 8;
342  } else if (cardCapacityMB <= 128) {
343  sectorsPerCluster = 16;
344  } else if (cardCapacityMB <= 1024) {
345  sectorsPerCluster = 32;
346  } else if (cardCapacityMB <= 32768) {
347  sectorsPerCluster = 64;
348  } else {
349  // SDXC cards
350  sectorsPerCluster = 128;
351  }
352 
353  // set fake disk geometry
354  sectorsPerTrack = cardCapacityMB <= 256 ? 32 : 63;
355 
356  if (cardCapacityMB <= 16) {
357  numberOfHeads = 2;
358  } else if (cardCapacityMB <= 32) {
359  numberOfHeads = 4;
360  } else if (cardCapacityMB <= 128) {
361  numberOfHeads = 8;
362  } else if (cardCapacityMB <= 504) {
363  numberOfHeads = 16;
364  } else if (cardCapacityMB <= 1008) {
365  numberOfHeads = 32;
366  } else if (cardCapacityMB <= 2016) {
367  numberOfHeads = 64;
368  } else if (cardCapacityMB <= 4032) {
369  numberOfHeads = 128;
370  } else {
371  numberOfHeads = 255;
372  }
373 
374  // Erase all data on card (TRIM)
375  uint32_t const ERASE_SIZE = 262144L;
376  uint32_t firstBlock = 0;
377  uint32_t lastBlock;
378  do {
379  lastBlock = firstBlock + ERASE_SIZE - 1;
380  if (lastBlock >= cardSizeBlocks) {
381  lastBlock = cardSizeBlocks - 1;
382  }
383  if (!card->erase(firstBlock, lastBlock)) {
384  return false; // Erase fail
385  }
386  delay(0); // yield to the OS to avoid WDT
387  firstBlock += ERASE_SIZE;
388  } while (firstBlock < cardSizeBlocks);
389 
390  if (!card->readBlock(0, cache->data)) {
391  return false;
392  }
393 
394  if (card->type() != sdfat::SD_CARD_TYPE_SDHC) {
395  return makeFat16();
396  } else {
397  return makeFat32();
398  }
399  }
