66 inline size_t split_args(char *line, char **argv, size_t argv_size)
67 {
68  const int QUOTE = '"';
69  const int ESCAPE = '\\';
70  const int SPACE = ' ';
71  split_state_t state = SS_SPACE;
72  int argc = 0;
73  char *next_arg_start = line;
74  char *out_ptr = line;
75  for (char *in_ptr = line; argc < argv_size - 1; ++in_ptr) {
76  int char_in = (unsigned char) *in_ptr;
77  if (char_in == 0) {
78  break;
79  }
80  int char_out = -1;
81 
82  switch (state) {
83  case SS_SPACE:
84  if (char_in == SPACE) {
85  /* skip space */
86  } else if (char_in == QUOTE) {
87  next_arg_start = out_ptr;
88  state = SS_QUOTED_ARG;
89  } else if (char_in == ESCAPE) {
90  next_arg_start = out_ptr;
91  state = SS_ARG_ESCAPED;
92  } else {
93  next_arg_start = out_ptr;
94  state = SS_ARG;
95  char_out = char_in;
96  }
97  break;
98 
99  case SS_QUOTED_ARG:
100  if (char_in == QUOTE) {
101  END_ARG();
102  } else if (char_in == ESCAPE) {
103  state = SS_QUOTED_ARG_ESCAPED;
104  } else {
105  char_out = char_in;
106  }
107  break;
108 
109  case SS_ARG_ESCAPED:
110  case SS_QUOTED_ARG_ESCAPED:
111  if (char_in == ESCAPE || char_in == QUOTE || char_in == SPACE) {
112  char_out = char_in;
113  } else {
114  /* unrecognized escape character, skip */
115  }
116  state = (split_state_t) (state & (~SS_FLAG_ESCAPE));
117  break;
118 
119  case SS_ARG:
120  if (char_in == SPACE) {
121  END_ARG();
122  } else if (char_in == ESCAPE) {
123  state = SS_ARG_ESCAPED;
124  } else {
125  char_out = char_in;
126  }
127  break;
128  }
129  /* need to output anything? */
130  if (char_out >= 0) {
131  *out_ptr = char_out;
132  ++out_ptr;
133  }
134  }
135  /* make sure the final argument is terminated */
136  *out_ptr = 0;
137  /* finalize the last argument */
138  if (state != SS_SPACE && argc < argv_size - 1) {
139  argv[argc++] = next_arg_start;
140  }
141  /* add a NULL at the end of argv */
142  argv[argc] = NULL;
143

145 }
