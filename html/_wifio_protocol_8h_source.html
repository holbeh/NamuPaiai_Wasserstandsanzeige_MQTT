118 {
119  // TODO: calculate parity
120  t.parity = 1;
121  if (twi_writeTo(addr, reinterpret_cast<uint8_t*>(&t), sizeof(T), true) != sizeof(T))

127 template<typename TC, typename TR>
128 bool sendCommandWaitForReply(uint8_t addr, TC& c, Command rt, TR& r, int32_t d_us)
129 {
130  c.parity = 1;
131  if (twi_writeTo(addr, reinterpret_cast<uint8_t*>(&c), sizeof(TC), true) != sizeof(TC))
132  {
133  protocolError();
134  return false;
135  }
136  if (d_us)
137  {
138  delayMicroseconds(d_us);
139  }
140  if (twi_readFrom(addr, reinterpret_cast<uint8_t*>(&r), sizeof(TR), true) != sizeof(TR) || r.header.cmd != rt)
141  {
142  protocolError();
143  return false;
144  }
145  return true;
146 }
