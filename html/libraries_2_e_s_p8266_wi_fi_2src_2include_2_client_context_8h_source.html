302  bool wait_until_sent(int max_wait_ms = WIFICLIENT_MAX_FLUSH_WAIT_MS)
303  {
304  // https://github.com/esp8266/Arduino/pull/3967#pullrequestreview-83451496
305  // option 1 done
306  // option 2 / _write_some() not necessary since _datasource is always nullptr here
307 
308  if (!_pcb)
309  return true;
310 
311  int prevsndbuf = -1;
312 
313  // wait for peer's acks to flush lwIP's output buffer
314  uint32_t last_sent = millis();
315  while (1) {
316  if (millis() - last_sent > (uint32_t) max_wait_ms) {
317 #ifdef DEBUGV
318  // wait until sent: timeout
319  DEBUGV(":wustmo\n");
320 #endif
321  // All data was not flushed, timeout hit
322  return false;
323  }
324 
325  // force lwIP to send what can be sent
326  tcp_output(_pcb);
327 
328  int sndbuf = tcp_sndbuf(_pcb);
329  if (sndbuf != prevsndbuf) {
330  // send buffer has changed (or first iteration)
331  prevsndbuf = sndbuf;
332  // We just sent a bit, move timeout forward
333  last_sent = millis();
334  }
335 
336  delay(0); // from sys or os context
337 
338  if ((state() != ESTABLISHED) || (sndbuf == TCP_SND_BUF)) {
339  break;
340  }
341  }
342 
343  // All data flushed
344  return true;
345  }

437  size_t _write_from_source(DataSource* ds)
438  {
439  assert(_datasource == nullptr);
440  assert(!_send_waiting);
441  _datasource = ds;
442  _written = 0;
443  _op_start_time = millis();
444  do {
445  if (_write_some()) {
446  _op_start_time = millis();
447  }
448 
449  if (!_datasource->available() || _is_timeout() || state() == CLOSED) {
450  if (_is_timeout()) {
451  DEBUGV(":wtmo\r\n");
452  }
453  delete _datasource;
454  _datasource = nullptr;
455  break;
456  }
457 
458  _send_waiting = true;
459  // This delay will be interrupted by esp_schedule on next received ack
460  delay(_timeout_ms);
461  } while(true);
462  _send_waiting = false;
463 
464  if (_sync)
465  wait_until_sent();
466 
467  return _written;
468  }

470  bool _write_some()
471  {
472  if (!_datasource || !_pcb) {
473  return false;
474  }
475 
476  DEBUGV(":wr %d %d\r\n", _datasource->available(), _written);
477 
478  bool has_written = false;
479 
480  while (_datasource) {
481  if (state() == CLOSED)
482  return false;
483  size_t next_chunk_size = std::min((size_t)tcp_sndbuf(_pcb), _datasource->available());
484  if (!next_chunk_size)
485  break;
486  const uint8_t* buf = _datasource->get_buffer(next_chunk_size);
487 
488  uint8_t flags = 0;
489  if (next_chunk_size < _datasource->available())
490  // PUSH is meant for peer, telling to give data to user app as soon as received
491  // PUSH "may be set" when sender has finished sending a "meaningful" data block
492  // PUSH does not break Nagle
493  // #5173: windows needs this flag
494  // more info: https://lists.gnu.org/archive/html/lwip-users/2009-11/msg00018.html
495  flags |= TCP_WRITE_FLAG_MORE; // do not tcp-PuSH (yet)
496  if (!_sync)
497  // user data must be copied when data are sent but not yet acknowledged
498  // (with sync, we wait for acknowledgment before returning to user)
499  flags |= TCP_WRITE_FLAG_COPY;
500 
501  err_t err = tcp_write(_pcb, buf, next_chunk_size, flags);
502 
503  DEBUGV(":wrc %d %d %d\r\n", next_chunk_size, _datasource->available(), (int)err);
504 
505  if (err == ERR_OK) {
506  _datasource->release_buffer(buf, next_chunk_size);
507  _written += next_chunk_size;
508  has_written = true;
509  } else {
510  // ERR_MEM(-1) is a valid error meaning
511  // "come back later". It leaves state() opened
512  break;
513  }
514  }
515 
516  if (has_written)
517  {
518  // lwIP's tcp_output doc: "Find out what we can send and send it"
519  // *with respect to Nagle*
520  // more info: https://lists.gnu.org/archive/html/lwip-users/2017-11/msg00134.html
521  tcp_output(_pcb);
522  }
523 
524  return has_written;
525  }

544  void _consume(size_t size)
545  {
546  if(_pcb)
547  tcp_recved(_pcb, size);
548  ptrdiff_t left = _rx_buf->len - _rx_buf_offset - size;
549  if(left > 0) {
550  _rx_buf_offset += size;
551  } else if(!_rx_buf->next) {
552  DEBUGV(":c0 %d, %d\r\n", size, _rx_buf->tot_len);
553  pbuf_free(_rx_buf);
554  _rx_buf = 0;
555  _rx_buf_offset = 0;
556  } else {
557  DEBUGV(":c %d, %d, %d\r\n", size, _rx_buf->len, _rx_buf->tot_len);
558  auto head = _rx_buf;
559  _rx_buf = _rx_buf->next;
560  _rx_buf_offset = 0;
