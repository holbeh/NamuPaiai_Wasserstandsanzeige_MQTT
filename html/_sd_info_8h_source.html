480 inline uint32_t sdCardCapacity(csd_t* csd) {
481  if (csd->v1.csd_ver == 0) {
482  uint8_t read_bl_len = csd->v1.read_bl_len;
483  uint16_t c_size = (csd->v1.c_size_high << 10)
484  | (csd->v1.c_size_mid << 2) | csd->v1.c_size_low;
485  uint8_t c_size_mult = (csd->v1.c_size_mult_high << 1)
486  | csd->v1.c_size_mult_low;
487  return (uint32_t)(c_size + 1) << (c_size_mult + read_bl_len - 7);
488  } else if (csd->v2.csd_ver == 1) {
489  uint32_t c_size = 0X10000L * csd->v2.c_size_high + 0X100L
490  * (uint32_t)csd->v2.c_size_mid + csd->v2.c_size_low;
491  return (c_size + 1) << 10;
492  } else {
493  return 0;
494  }
