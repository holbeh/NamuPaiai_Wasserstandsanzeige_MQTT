46 /*
47  * Maximum size for a RSA modulus (in bits). Allocated stack buffers
48  * depend on that size, so this value should be kept small. Currently,
49  * 2048-bit RSA keys offer adequate security, and should still do so for
50  * the next few decades; however, a number of widespread PKI have
51  * already set their root keys to RSA-4096, so we should be able to
52  * process such keys.
53  *
54  * This value MUST be a multiple of 64. This value MUST NOT exceed 47666
55  * (some computations in RSA key generation rely on the factor size being
56  * no more than 23833 bits). RSA key sizes beyond 3072 bits don't make a
57  * lot of sense anyway.
58  */

791 /*
792  * Comparison: returns 1 if x > y, 0 otherwise.
793  */
794 static inline uint32_t
795 GT(uint32_t x, uint32_t y)
796 {
797  /*
798  * If both x < 2^31 and x < 2^31, then y-x will have its high
799  * bit set if x > y, cleared otherwise.
800  *
801  * If either x >= 2^31 or y >= 2^31 (but not both), then the
802  * result is the high bit of x.
803  *
804  * If both x >= 2^31 and y >= 2^31, then we can virtually
805  * subtract 2^31 from both, and we are back to the first case.
806  * Since (y-2^31)-(x-2^31) = y-x, the subtraction is already
807  * fine.
808  */
809  uint32_t z;
810 
811  z = y - x;
812  return (z ^ ((x ^ y) & (x ^ z))) >> 31;
813 }

1739 /*
1740  * The constant-time implementation is "bitsliced": the 128-bit state is
1741  * split over eight 32-bit words q* in the following way:
1742  *
1743  * -- Input block consists in 16 bytes:
1744  * a00 a10 a20 a30 a01 a11 a21 a31 a02 a12 a22 a32 a03 a13 a23 a33
1745  * In the terminology of FIPS 197, this is a 4x4 matrix which is read
1746  * column by column.
1747  *
1748  * -- Each byte is split into eight bits which are distributed over the
1749  * eight words, at the same rank. Thus, for a byte x at rank k, bit 0
1750  * (least significant) of x will be at rank k in q0 (if that bit is b,
1751  * then it contributes "b << k" to the value of q0), bit 1 of x will be
1752  * at rank k in q1, and so on.
1753  *
1754  * -- Ranks given to bits are in "row order" and are either all even, or
1755  * all odd. Two independent AES states are thus interleaved, one using
1756  * the even ranks, the other the odd ranks. Row order means:
1757  * a00 a01 a02 a03 a10 a11 a12 a13 a20 a21 a22 a23 a30 a31 a32 a33
1758  *
1759  * Converting input bytes from two AES blocks to bitslice representation
1760  * is done in the following way:
1761  * -- Decode first block into the four words q0 q2 q4 q6, in that order,
1762  * using little-endian convention.
1763  * -- Decode second block into the four words q1 q3 q5 q7, in that order,
1764  * using little-endian convention.
1765  * -- Call br_aes_ct_ortho().
1766  *
1767  * Converting back to bytes is done by using the reverse operations. Note
1768  * that br_aes_ct_ortho() is its own inverse.
1769  */
1770 
1771 /*
1772  * Perform bytewise orthogonalization of eight 32-bit words. Bytes
1773  * of q0..q7 are spread over all words: for a byte x that occurs
1774  * at rank i in q[j] (byte x uses bits 8*i to 8*i+7 in q[j]), the bit
1775  * of rank k in x (0 <= k <= 7) goes to q[k] at rank 8*i+j.
1776  *
1777  * This operation is an involution.
1778  */
