231  bool next()
232  {
233  if (!_rx_buf)
234  return false;
235  if (!_first_buf_taken)
236  {
237  _first_buf_taken = true;
238  return true;
239  }
240 
241  auto deleteme = _rx_buf;
242  _rx_buf = _rx_buf->next;
243 
244  if (_rx_buf)
245  {
246  // we have interleaved informations on addresses within reception pbuf chain:
247  // before: (data-pbuf) -> (data-pbuf) -> (data-pbuf) -> ... in the receiving order
248  // now: (address-info-pbuf -> data-pbuf) -> (address-info-pbuf -> data-pbuf) -> ...
249 
250  // so the first rx_buf contains an address helper,
251  // copy it to "current address"
252  auto helper = (AddrHelper*)PBUF_ALIGNER(_rx_buf->payload);
253  _currentAddr = *helper;
254 
255  // destroy the helper in the about-to-be-released pbuf
256  helper->~AddrHelper();
257 
258  // forward in rx_buf list, next one is effective data
259  // current (not ref'ed) one will be pbuf_free'd with deleteme
260  _rx_buf = _rx_buf->next;
261 
262  // this rx_buf is not nullptr by construction,
263  // ref'ing it to prevent release from the below pbuf_free(deleteme)
264  pbuf_ref(_rx_buf);
265  }
266  pbuf_free(deleteme);
267 
268  _rx_buf_offset = 0;
269  return _rx_buf != nullptr;
270  }

314  size_t append(const char* data, size_t size)
315  {
316  if (!_tx_buf_head || _tx_buf_head->tot_len < _tx_buf_offset + size)
317  {
318  _reserve(_tx_buf_offset + size);
319  }
320  if (!_tx_buf_head || _tx_buf_head->tot_len < _tx_buf_offset + size)
321  {
322  DEBUGV("failed _reserve");
323  return 0;
324  }
325 
326  size_t left_to_copy = size;
327  while(left_to_copy)
328  {
329  // size already used in current pbuf
330  size_t used_cur = _tx_buf_offset - (_tx_buf_head->tot_len - _tx_buf_cur->tot_len);
331  size_t free_cur = _tx_buf_cur->len - used_cur;
332  if (free_cur == 0)
333  {
334  _tx_buf_cur = _tx_buf_cur->next;
335  continue;
336  }
337  size_t will_copy = (left_to_copy < free_cur) ? left_to_copy : free_cur;
338  memcpy(reinterpret_cast<char*>(_tx_buf_cur->payload) + used_cur, data, will_copy);
339  _tx_buf_offset += will_copy;
340  left_to_copy -= will_copy;
341  data += will_copy;
342  }
343  return size;
344  }

347  {

350  if(!tx_copy){

355  for (pbuf* p = _tx_buf_head; p; p = p->next) {

395  void _reserve(size_t size)
396  {
397  const size_t pbuf_unit_size = 128;
398  if (!_tx_buf_head)
399  {
400  _tx_buf_head = pbuf_alloc(PBUF_TRANSPORT, pbuf_unit_size, PBUF_RAM);
401  if (!_tx_buf_head)
402  {
403  return;
404  }
405  _tx_buf_cur = _tx_buf_head;
406  _tx_buf_offset = 0;
407  }
408 
409  size_t cur_size = _tx_buf_head->tot_len;
410  if (size < cur_size)
411  return;
412 
413  size_t grow_size = size - cur_size;
414 
415  while(grow_size)
416  {
417  pbuf* pb = pbuf_alloc(PBUF_TRANSPORT, pbuf_unit_size, PBUF_RAM);
418  if (!pb)
419  {
420  return;
421  }
422  pbuf_cat(_tx_buf_head, pb);
423  if (grow_size < pbuf_unit_size)
424  return;

437  void _recv(udp_pcb *upcb, pbuf *pb,
438  const ip_addr_t *srcaddr, u16_t srcport)
439  {
440  (void) upcb;
441 
442 #if LWIP_VERSION_MAJOR == 1
443  #define TEMPDSTADDR (&current_iphdr_dest)
444 #else
445  #define TEMPDSTADDR (ip_current_dest_addr())
446 #endif
447 
448  // chain this helper pbuf first
449  if (_rx_buf)
450  {
451  // there is some unread data
452  // chain pbuf
453 
454  // Addresses/ports are stored from this callback because lwIP's
455  // macro are valid only now.
456  //
457  // When peeking data from before payload start (like it was done
458  // before IPv6), there's no easy way to safely guess whether
459  // packet is from v4 or v6.
460  //
461  // Now storing data in an intermediate chained pbuf containing
462  // AddrHelper
463 
464  // allocate new pbuf to store addresses/ports
465  pbuf* pb_helper = pbuf_alloc(PBUF_RAW, sizeof(AddrHelper) + PBUF_ALIGNER_ADJUST, PBUF_RAM);
466  if (!pb_helper)
467  {
468  // memory issue - discard received data
469  pbuf_free(pb);
470  return;
471  }
472  // construct in place
473  new(PBUF_ALIGNER(pb_helper->payload)) AddrHelper(srcaddr, TEMPDSTADDR, srcport);
474  // chain it
475  pbuf_cat(_rx_buf, pb_helper);
476 
477  // now chain the new data pbuf
478  DEBUGV(":urch %d, %d\r\n", _rx_buf->tot_len, pb->tot_len);
479  pbuf_cat(_rx_buf, pb);
480  }
481  else
482  {
483  _currentAddr.srcaddr = srcaddr;
484  _currentAddr.dstaddr = TEMPDSTADDR;
485  _currentAddr.srcport = srcport;
486 
487  DEBUGV(":urn %d\r\n", pb->tot_len);
488  _first_buf_taken = false;
489  _rx_buf = pb;
490  _rx_buf_offset = 0;
491  }
492 
493  if (_on_rx) {
494  _on_rx();
495  }
496 
497  #undef TEMPDSTADDR
498 
499  }
