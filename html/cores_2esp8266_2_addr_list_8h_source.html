150 class AddressListIterator
151 {
152 public:
153  AddressListIterator (const netifWrapper& o) : netIf(o) {}
154  AddressListIterator (netif* netif) : netIf(netif)
155  {
156  // This constructor is called with lwIP's global netif_list, or
157  // nullptr. operator++() is designed to loop through _configured_
158  // addresses. That's why netIf's _num is initialized to -1 to allow
159  // returning the first usable address to AddressList::begin().
160  (void)operator++();
161  }
162 
163  const netifWrapper& operator* () const { return netIf; }
164  const netifWrapper* operator-> () const { return &netIf; }
165 
166  bool operator== (AddressListIterator& o) { return netIf.equal(*o); }
167  bool operator!= (AddressListIterator& o) { return !netIf.equal(*o); }
168 
169  AddressListIterator& operator= (const AddressListIterator& o) { netIf = o.netIf; return *this; }
170 
171  AddressListIterator operator++ (int)
172  {
173  AddressListIterator ret = *this;
174  (void)operator++();
175  return ret;
176  }
177 
178  AddressListIterator& operator++ ()
179  {
180  while (netIf._netif)
181  {
182  if (++netIf._num == IF_NUM_ADDRESSES)
183  {
184  // all addresses from current interface were iterated,
185  // switching to next interface
186  netIf = netifWrapper(netIf._netif->next);
187  continue;
188  }
189  if (!ip_addr_isany(netIf.ipFromNetifNum()))
190  // found an initialized address
191  break;
192  }
193  return *this;
194  }
195 
196  netifWrapper netIf;
197 };
