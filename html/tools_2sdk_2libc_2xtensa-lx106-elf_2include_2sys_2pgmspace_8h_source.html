85 __attribute__((optimize("-O3"), always_inline)) static inline uint32_t pgm_read_dword_unaligned(const void *addr) {
86  if (!(((int)addr)&3)) return *(const uint32_t *)addr;
87  int off = (((int)addr) & 3) << 3;
