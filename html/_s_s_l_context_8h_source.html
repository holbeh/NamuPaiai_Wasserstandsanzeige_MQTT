183  int read(uint8_t* dst, size_t size)
184  {
185  if (!_available) {
186  if (!_readAll()) {
187  return 0;
188  }
189  }
190  size_t will_copy = (_available < size) ? _available : size;
191  memcpy(dst, _read_ptr, will_copy);
192  _read_ptr += will_copy;
193  _available -= will_copy;
194  if (_available == 0) {
195  _read_ptr = nullptr;
196  /* Send pending outgoing data, if any */
197  if (_hasWriteBuffers()) {
198  _writeBuffersSend();
199  }
200  }
201  return will_copy;
202  }
