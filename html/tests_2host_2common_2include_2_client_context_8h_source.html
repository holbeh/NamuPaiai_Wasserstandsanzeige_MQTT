237  size_t write(Stream& stream)
238  {
239  size_t avail = stream.available();
240  uint8_t buf [avail];
241  avail = stream.readBytes(buf, avail);

244  while (avail && _sock >= 0)
245  {
246  size_t wrote = write(w, avail);
247  w += wrote;
